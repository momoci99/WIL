문제 개요

- 단순 문자열 치환 문제

전개
none

```js
function solution(s) {
  var answer = 0;
  const replaced = s
    .replaceAll("zero", 0)
    .replaceAll("one", 1)
    .replaceAll("two", 2)
    .replaceAll("three", 3)
    .replaceAll("four", 4)
    .replaceAll("five", 5)
    .replaceAll("six", 6)
    .replaceAll("seven", 7)
    .replaceAll("eight", 8)
    .replaceAll("nine", 9);

  return parseInt(replaced, 10);
}
```
